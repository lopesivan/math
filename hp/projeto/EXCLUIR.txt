%%HP: T(3)A(R)F(.);
\<<
@ Exclui um backup
@ Recebe o nome dele

:3:"BACKUPS/INFO" RCL \-> nome info \<<
@ Busca a posição dele na lista e o exclui
IFERR
	1. info SIZE FOR j
		IF info j GET 2. GET nome SAME THEN
			@ Encontrou: exclui e sai do loop
			info 1. j 1. - SUB
			info j 1. + info SIZE SUB +
			:3:"BACKUPS/INFO" PURGE
			:3:"BACKUPS/INFO" STO
			0 DOERR
		END
	NEXT
THEN END

@ Exclui do SD
nome PURGE
\>>

@ Volta para o menu anterior
:3:"BACKUPS/MOSTRAR" EVAL
\>>
